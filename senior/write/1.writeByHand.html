# 简单实现一个webpack

# 1.写一个打包案例

# 1.文件

src/base/b.js

module.exports = "b"
1

src/a.js

let b = require("./base/b.js")
module.exports = "a" + b

src/index.js

let str = require("./a.js")

# 2.配置

webpack.config.js

let path = require("path")
module.exports = {
  mode: "development",
  entry: "./src/index.js",
  output: {
    filename: "bundle.js",
    path: path.resolve(__dirname, "dist"),
  },
}
