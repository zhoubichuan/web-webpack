## 安装

```
yarn add babel babel-loader @babel/core @babel/preset-env -D
```

## webpack.config.js

```
{
    test: /\.js$/,
    use: {
        loader: "babel-loader",
        options: {
        presets: ["@babel/preset-env"]
        }
    }
}
```

## index.js

```
let fn = () => {
  console.log("log");
};
fn();
```

## 安装

```
yarn add @babel/plugin-proposal-class-properties -D
```

## webpack.config.js

```
{
    test: /\.js$/,
    use: {
        loader: "babel-loader",
        options: {
        presets: ["@babel/preset-env"],
        plugins: ["@babel/plugin-proposal-class-properties"]
        }
    }
}
```
